set(SOURCES
        wraphelper.cpp wraphelper.h
        wrapnvml.h wrapnvml.cpp
        wrapadl.h wrapadl.cpp

        )

add_library(hwmon ${SOURCES})

endif ()

    find_package(CUDA REQUIRED)
    target_include_directories(hwmon PUBLIC ${CUDA_INCLUDE_DIRS})
